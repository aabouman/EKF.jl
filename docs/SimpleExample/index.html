Defining The Process and Measurement Functions

function process(xᵢ::AbstractVector, uᵢ::AbstractVector)
    Δt = .1
    A = [1 Δt 0  0;
         0  1 0  0;
         0  0 1 Δt;
         0  0 0  1]
    return A * xᵢ
end

function measure(xᵢ::AbstractVector)
    x, ẋ, y, ẏ = xᵢ
    return [sqrt(x^2 + y^2); atan(y, x)]
end
